#include <iostream> 
#include <limits.h>
int main()
{
    const int size = 6;
    const int size1 = 5;
    int arr[size][size1];
    int n = 0;
    std::cin >> n;
    for (int i = 0; i < size; i++)
    {
        for (int j = 0; j < size1; j++)
        {
            std::cin >> arr[i][j];
        }
    }
    for (int i = 0; i < size; i++)
    {
        if (n - 1 == i)
            continue;
        for (int j = 0; j < size1; j++)
        {
            std::cout << arr[i][j] << " ";
        }
        std::cout << "\n";
    }
}
